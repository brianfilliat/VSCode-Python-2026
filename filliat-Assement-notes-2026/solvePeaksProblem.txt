def solvePeaksProblem(arr):
    n = len(arr)
    peaks = []
    for i in range(1, n-1):
        if arr[i] > arr[i-1] and arr[i] > arr[i+1]:
            peaks.append(i)
    if len(peaks) < 2:
        return 0
    max_diff = 0
    for i in range(1, len(peaks)):
        diff = peaks[i] - peaks[i-1]
        if diff > max_diff:
            max_diff = diff
    return max_diff

if __name__ == '__main__':
    n = int(input().strip())
    arr = list(map(int, input().strip().split()))
    result = solvePeaksProblem(arr)
    print(result)
