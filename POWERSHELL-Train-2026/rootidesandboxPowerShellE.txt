# Simple employee records manager for a small business
# Stores records in a script-scoped array and provides CRUD + utilities

# Initialization
if (-not (Test-Path -Path Variable:script:employeeRecords)) {
    $script:employeeRecords = @()
}

function New-EmployeeRecord {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)] [string] $Name,
        [Parameter(Mandatory=$true)] [string] $Position,
        [Parameter(Mandatory=$true)] [string] $Department,
        [Parameter(Mandatory=$true)] [decimal] $Salary
    )

    $employee = [PSCustomObject]@{
        Name       = $Name
        Position   = $Position
        Department = $Department
        Salary     = [decimal]$Salary
    }

    return $employee
}

function Add-EmployeeRecord {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)] [string] $Name,
        [Parameter(Mandatory=$true)] [string] $Position,
        [Parameter(Mandatory=$true)] [string] $Department,
        [Parameter(Mandatory=$true)] [decimal] $Salary
    )

    if (-not (Test-Path -Path Variable:script:employeeRecords)) { $script:employeeRecords = @() }

    $newRecord = New-EmployeeRecord -Name $Name -Position $Position -Department $Department -Salary $Salary
    $script:employeeRecords += $newRecord
    return $newRecord
}

function Remove-EmployeeRecord {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)] [string] $Name
    )

    if (-not (Test-Path -Path Variable:script:employeeRecords) -or $script:employeeRecords.Count -eq 0) {
        Write-Verbose "No employee records to remove from."
        return $null
    }

    $matches = $script:employeeRecords | Where-Object { $_.Name -ieq $Name }
    if (-not $matches) {
        return $null
    }

    foreach ($m in $matches) {
        $script:employeeRecords = $script:employeeRecords | Where-Object { $_ -ne $m }
    }

    return $matches
}

function Edit-EmployeeRecord {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true, Position=0)] [string] $Name,
        [Parameter(Mandatory=$false)] [string] $NewName,
        [Parameter(Mandatory=$false)] [string] $NewPosition,
        [Parameter(Mandatory=$false)] [string] $NewDepartment,
        [Parameter(Mandatory=$false)] [decimal] $NewSalary
    )

    if (-not (Test-Path -Path Variable:script:employeeRecords) -or $script:employeeRecords.Count -eq 0) {
        Write-Verbose "No employee records to edit."
        return $null
    }

    $found = $script:employeeRecords | Where-Object { $_.Name -ieq $Name }
    if (-not $found) { return $null }

    $updated = @()
    foreach ($record in $found) {
        if ($PSBoundParameters.ContainsKey('NewName') -and $NewName) {
            $record.Name = $NewName
        }
        if ($PSBoundParameters.ContainsKey('NewPosition') -and $NewPosition) {
            $record.Position = $NewPosition
        }
        if ($PSBoundParameters.ContainsKey('NewDepartment') -and $NewDepartment) {
            $record.Department = $NewDepartment
        }
        if ($PSBoundParameters.ContainsKey('NewSalary')) {
            $record.Salary = [decimal]$NewSalary
        }
        $updated += $record
    }

    return $updated
}

function List-EmployeeRecords {
    [CmdletBinding()]
    param()

    if (-not (Test-Path -Path Variable:script:employeeRecords)) { $script:employeeRecords = @() }

    foreach ($e in $script:employeeRecords) {
        [PSCustomObject]@{
            Name       = $e.Name
            Position   = $e.Position
            Department = $e.Department
            Salary     = $e.Salary
        }
    }
}

function Find-EmployeeRecord {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)] [string] $Name
    )

    if (-not (Test-Path -Path Variable:script:employeeRecords) -or $script:employeeRecords.Count -eq 0) {
        Write-Output "No employee records available."
        return $null
    }

    $matches = $script:employeeRecords | Where-Object { $_.Name -like "*${Name}*" -or $_.Name -ieq $Name }
    if ($matches) {
        return $matches
    }

    Write-Output "Employee '$Name' not found."
    return $null
}

function Clear-EmployeeRecords {
    [CmdletBinding()]
    param()

    $script:employeeRecords = @()
    return $true
}

function Get-AverageSalary {
    [CmdletBinding()]
    param()

    if (-not (Test-Path -Path Variable:script:employeeRecords) -or $script:employeeRecords.Count -eq 0) {
        return 0
    }

    $total = ($script:employeeRecords | Measure-Object -Property Salary -Sum).Sum
    $count = $script:employeeRecords.Count
    if ($count -eq 0) { return 0 }
    return [decimal]($total / $count)
}
